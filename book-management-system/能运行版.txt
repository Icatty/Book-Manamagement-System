 int check_data_out()//check文件写入数组    函数 
 {	
	int r=-1;
	fp=fopen("备份.txt","r+");
	if(fp==NULL)
	{
		printf("打开失败");
	}
	for(int i=0;i<=100;i++)
	{	
		if(i%3==0)
		{
			r=r+1;
		}
		if(feof(fp)!=0)
		{	
			end=i;
			break;
		}
		if(i%3==0)
		{	
			fscanf(fp,"%s",s[r].book);
		}
		if((i-1)%3==0)
		{	 
			fscanf(fp,"%s",s[r].writer);
		}
		if((i-2)%3==0)
		{	
			fscanf(fp,"%s",s[r].num);
		}
	}
	fclose(fp);
 }

int b_return() //还书函数 
{		
	char ONE[10];
	strcpy(ONE,"1");
	int borrow_num=0,choose;	
	for(int i=0;i<100;i++)
	{	
		if(i==(end-1)/3)
		{
			break;
		}	 	
		if(strcmp(s[i].num,stu_num)==0)//账号不能是"1"
		{
			printf("借阅记录：  %s\n",s[i].book);
			borrow_num++;
		}				
	}
	if(borrow_num==0)
	{
		printf("您没有借阅记录\n");
	}
	while(borrow_num!=0)
	{	
		printf("请选择  1、 还书     2、退出\n");
		scanf("%d",&choose);
		if(choose==1)
		{	int b=b_check();		
			if(b==0)
			{
				printf("您搜索的书不存在哟\n");
			}
			if(strcmp(s[b].num,"1")==0)
			{	
				printf("该书没有借阅记录\n");
			}
			if(strcmp(s[b].num,stu_num)==0)
			{	
				strcpy(s[b].num,"1");
				printf("你已归还该书\n");
			}
			if(strcmp(s[b].num,ONE)!=0&&strcmp(s[b].num,stu_num)!=0&&b!=0)
			{	
				printf("您没有借过该书\n");
			}															
		}
		if(choose!=1)
		{
			break;
		}
	}
	printf("你已退出还书功能\n");
}
